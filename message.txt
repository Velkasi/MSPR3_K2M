-- Schéma : core
CREATE TABLE core.application (
   Id_APPLICATION SERIAL,
   Version_OS_Appli VARCHAR(50),
   PRIMARY KEY(Id_APPLICATION)
);

CREATE TABLE core.technicien (
   Id_TECHNICIEN SERIAL,
   Nom_technicien VARCHAR(100),
   Prenom_technicien VARCHAR(100),
   PRIMARY KEY(Id_TECHNICIEN)
);

CREATE TABLE core.type_disques (
   Id_TYPE_DISQUES SERIAL,
   Type_disques_clients VARCHAR(50),
   PRIMARY KEY(Id_TYPE_DISQUES)
);

CREATE TABLE core.taille_disques (
   Id_TAILLE_DISQUES SERIAL,
   Taille_disques_client VARCHAR(50),
   PRIMARY KEY(Id_TAILLE_DISQUES)
);

CREATE TABLE core.type_os (
   Id_TYPE_OS SERIAL,
   Type_d_OS VARCHAR(50),
   Version_OS VARCHAR(50),
   PRIMARY KEY(Id_TYPE_OS)
);

CREATE TABLE core.type_ram (
   Id_TYPE_RAM SERIAL,
   Nb_RAM VARCHAR(50),
   PRIMARY KEY(Id_TYPE_RAM)
);

CREATE TABLE core.personne (
   Id_Personne SERIAL,
   Nom VARCHAR(50),
   Prenom VARCHAR(50),
   PRIMARY KEY(Id_Personne)
);

CREATE TABLE core.adresse (
   Id_ADRESSE SERIAL,
   Voie VARCHAR(50),
   N_Voie VARCHAR(50),
   CP VARCHAR(50),
   Ville VARCHAR(50),
   PRIMARY KEY(Id_ADRESSE)
);

-- Schéma : nfl_schema
CREATE TABLE nfl_schema.franchise (
   Id_FRANCHISE SERIAL,
   Nom_Franchise VARCHAR(50),
   Id_ADRESSE INTEGER NOT NULL,
   PRIMARY KEY(Id_FRANCHISE),
   FOREIGN KEY(Id_ADRESSE) REFERENCES core.adresse(Id_ADRESSE)
);

CREATE TABLE nfl_schema.info_collecte (
   Id_INFO_COLLECTE SERIAL,
   Adresse_MAC VARCHAR(50),
   Etat_instance BOOLEAN,
   Adresse_ip_client VARCHAR(50),
   CPU_Client VARCHAR(50),
   Adresse_ip_instance VARCHAR(50),
   NB_Requete_client VARCHAR(50),
   Id_TYPE_OS INTEGER NOT NULL,
   Id_TECHNICIEN INTEGER NOT NULL,
   PRIMARY KEY(Id_INFO_COLLECTE),
   UNIQUE(Id_TYPE_OS),
   FOREIGN KEY(Id_TYPE_OS) REFERENCES core.type_os(Id_TYPE_OS),
   FOREIGN KEY(Id_TECHNICIEN) REFERENCES core.technicien(Id_TECHNICIEN)
);

CREATE TABLE nfl_schema.licence (
   Id_LICENCE SERIAL,
   Numero_licences SMALLINT,
   Id_APPLICATION INTEGER NOT NULL,
   Id_INFO_COLLECTE INTEGER NOT NULL,
   PRIMARY KEY(Id_LICENCE),
   FOREIGN KEY(Id_APPLICATION) REFERENCES core.application(Id_APPLICATION),
   FOREIGN KEY(Id_INFO_COLLECTE) REFERENCES nfl_schema.info_collecte(Id_INFO_COLLECTE)
);

CREATE TABLE nfl_schema.date_collecte (
   Id_Date_collecte SERIAL,
   Id_INFO_COLLECTE INTEGER NOT NULL,
   Id_Personne INTEGER NOT NULL,
   PRIMARY KEY(Id_Date_collecte),
   FOREIGN KEY(Id_INFO_COLLECTE) REFERENCES nfl_schema.info_collecte(Id_INFO_COLLECTE),
   FOREIGN KEY(Id_Personne) REFERENCES core.personne(Id_Personne)
);

CREATE TABLE nfl_schema.fait_parti (
   Id_FRANCHISE INTEGER,
   Id_Personne INTEGER,
   PRIMARY KEY(Id_FRANCHISE, Id_Personne),
   FOREIGN KEY(Id_FRANCHISE) REFERENCES nfl_schema.franchise(Id_FRANCHISE),
   FOREIGN KEY(Id_Personne) REFERENCES client_a.client(Id_Personne)
);

CREATE TABLE nfl_schema.contient_type (
   Id_INFO_COLLECTE INTEGER,
   Id_TYPE_DISQUES INTEGER,
   PRIMARY KEY(Id_INFO_COLLECTE, Id_TYPE_DISQUES),
   FOREIGN KEY(Id_INFO_COLLECTE) REFERENCES nfl_schema.info_collecte(Id_INFO_COLLECTE),
   FOREIGN KEY(Id_TYPE_DISQUES) REFERENCES core.type_disques(Id_TYPE_DISQUES)
);

CREATE TABLE nfl_schema.contient_taille (
   Id_INFO_COLLECTE INTEGER,
   Id_TAILLE_DISQUES INTEGER,
   PRIMARY KEY(Id_INFO_COLLECTE, Id_TAILLE_DISQUES),
   FOREIGN KEY(Id_INFO_COLLECTE) REFERENCES nfl_schema.info_collecte(Id_INFO_COLLECTE),
   FOREIGN KEY(Id_TAILLE_DISQUES) REFERENCES core.taille_disques(Id_TAILLE_DISQUES)
);

CREATE TABLE nfl_schema.contient_ram (
   Id_INFO_COLLECTE INTEGER,
   Id_TYPE_RAM INTEGER,
   PRIMARY KEY(Id_INFO_COLLECTE, Id_TYPE_RAM),
   FOREIGN KEY(Id_INFO_COLLECTE) REFERENCES nfl_schema.info_collecte(Id_INFO_COLLECTE),
   FOREIGN KEY(Id_TYPE_RAM) REFERENCES core.type_ram(Id_TYPE_RAM)
);
